"use strict";
/**
 * 
 * @param {number} add ;
 * @param {number} x ;
 * @param {number} y ;
 * @return{number};
 */
function compute(add,x,y){
    return add(x,y);
}
/**
 * 
 * @param {number} x ;
 * @param {number} y ;
 * @return{number};
 */
function add(x,y){
    return x+y
}
let prompt = require("prompt-sync")();
let n = +prompt("Enter first number:");
let m = +prompt("Enter first number:");
console.log(`sum of the number using function expression:${compute(add,n,m)}`)
